 125    def safe_get(self, endpoint: str, timeout: float = 1.0) -> dict:
 126        """
 127        A helper method to safely perform GET requests.
 128
 129        Args:
 130            endpoint (str): The full URL or endpoint to send a GET request to.
 131            timeout (float): The request timeout in seconds.
 132
 133        Returns:
 134            dict: Parsed JSON response, or an empty dict if an error occurred.
 135        """
 136        try:
 137            response = requests.get(endpoint, timeout=timeout)
 138            response.raise_for_status()
 139            return response.json()
 140        except requests.RequestException as e:
 141            print(f"Error in GET request to {endpoint}: {e}")
 142            return {}

 126    def safe_get(self, endpoint: str, timeout: float = 1.0) -> dict:
 127        """
 128        A helper method to safely perform GET requests.
 129
 130        Args:
 131            endpoint (str): The full URL or endpoint to send a GET request to.
 132            timeout (float): The request timeout in seconds.
 133
 134        Returns:
 135            dict: Parsed JSON response, or an empty dict if an error occurred.
 136        """
 137        try:
 138            response = requests.get(endpoint, timeout=timeout)
 139            response.raise_for_status()
 140            return response.json()
 141        except requests.RequestException as e:
 142            print(f"Error in GET request to {endpoint}: {e}")
 143            return {}
