449    def set_player_tracker_overlay_type(self, overlay_type: str) -> dict:
450        """
451        Set the player tracker overlay type.
452
453        Args:
454            overlay_type (str): Overlay type ('DOT', 'TILE', 'HULL', 'OUTLINE')
455
456        Returns:
457            dict: Response from the API
458        """
459        endpoint = f"{self.base_url}player_tracker/set_type"
460
461        # Validate overlay type
462        valid_types = ["DOT", "TILE", "HULL", "OUTLINE"]
463        overlay_type = overlay_type.upper()
464        if overlay_type not in valid_types:
465            return {"error": f"Invalid overlay type '{overlay_type}'. Valid types: {', '.join(valid_types)}"}
466
467        data = {"type": overlay_type}
468
469        try:
470            response = requests.post(endpoint, json=data, timeout=self.timeout)
471            response.raise_for_status()
472            return response.json()
473        except requests.RequestException as e:
474            print(f"Error in set player tracker overlay type request: {e}")
475            return {"error": str(e)}
