 10    def safe_get(self, endpoint: str, timeout: float = 1.0) -> dict:
 11        """
 12        A helper method to safely perform GET requests.
 13
 14        Args:
 15            endpoint (str): The full URL or endpoint to send a GET request to.
 16            timeout (float): The request timeout in seconds.
 17
 18        Returns:
 19            dict: Parsed JSON response, or an empty dict if an error occurred.
 20        """
 21        try:
 22            response = requests.get(endpoint, timeout=timeout)
 23            response.raise_for_status()
 24            return response.json()
 25        except requests.RequestException as e:
 26            print(f"Error in GET request to {endpoint}: {e}")
 27            return {}

 32    def check_npcs_exist(self, npc_names: List[str] = None, npc_ids: List[int] = None) -> dict:
 33        """
 34        Checks if NPCs with the specified names or IDs exist in the current scene.
 35
 36        Args:
 37            npc_names (List[str], optional): List of NPC names to check for existence
 38            npc_ids (List[int], optional): List of NPC IDs to check for existence
 39
 40        Returns:
 41            dict: Dictionary with NPC names/IDs as keys and existence status as values
 42        """
 43        if not npc_names and not npc_ids:
 44            return {}
 45
 46        params = []
 47        if npc_names:
 48            if isinstance(npc_names, str):
 49                npc_names = [npc_names]
 50            params.append(f"names={','.join(npc_names)}")
 51
 52        if npc_ids:
 53            if isinstance(npc_ids, int):
 54                npc_ids = [npc_ids]
 55            params.append(f"ids={','.join(map(str, npc_ids))}")
 56
 57        endpoint = f"{self.base_url}npcs/exist?{'&'.join(params)}"
 58
 59        data = self.safe_get(endpoint)
 60        if data.get("error") or data.get("state"):
 61            return {}
 62
 63        # For single NPC ID checks, return True if the NPC exists
 64        if isinstance(npc_ids, list) and len(npc_ids) == 1:
 65            npc_data = data.get(str(npc_ids[0]))
 66            if npc_data and isinstance(npc_data, dict):
 67                return npc_data.get("npcExists") == "True"
 68            return False
 69
 70        return data

270    def get_nearest_npc_world_location(self, npc_names: List[str] = None, npc_ids: List[int] = None) -> tuple:
271        """
272        Gets the world location (x, y, z) of the nearest NPC.
273
274        Args:
275            npc_names (List[str], optional): List of NPC names to search for
276            npc_ids (List[int], optional): List of NPC IDs to search for
277
278        Returns:
279            tuple: (x, y, z) world coordinates of the nearest NPC, or None if not found
280        """
281        npc_data = self.get_nearest_npc(npc_names, npc_ids)
282
283        if not npc_data or npc_data.get("error") or npc_data.get("state"):
284            return None
285
286        world_location = npc_data.get("npcWorldLocation", {})
287        if not world_location:
288            return None
289
290        x = world_location.get("locationX")
291        y = world_location.get("locationY")
292        z = world_location.get("locationZ")
293
294        if x is not None and y is not None and z is not None:
295            return (int(x), int(y), int(z))
296
297        return None

299    def interact_with_npc(self, action: str) -> dict:
300        """
301        Interacts with the currently selected/invokable NPC using the specified action.
302
303        Args:
304            action (str): The action to perform on the NPC (e.g., "attack", "talk-to", "pickpocket", etc.)
305
306        Returns:
307            dict: Response from the API indicating success or failure
308        """
309        endpoint = f"{self.base_url}invokes/npcs/interact"
310
311        # Always consume clicks by default
312        data = {"action": action, "consume": "true"}
313
314        try:
315            response = requests.post(endpoint, json=data, timeout=3)
316            response.raise_for_status()
317            return response.json()
318        except requests.RequestException as e:
319            print(f"Error in NPC interaction request: {e}")
320            return {"error": str(e)}

322    def is_any_npc_interacting_with_me(self, npc_ids: List[int] = None) -> bool:
323        """
324        Checks if any NPC is currently interacting with the player.
325
326        Args:
327            npc_ids (List[int], optional): List of specific NPC IDs to check. If None, uses common attackable NPCs.
328
329        Returns:
330            bool: True if any NPC is interacting with the player, False otherwise
331        """
332        if npc_ids:
333            # Check specific NPC IDs provided by the user
334            if isinstance(npc_ids, int):
335                npc_ids = [npc_ids]
336
337            for npc_id in npc_ids:
338                endpoint = f"{self.base_url}npcs/nearest_my_interacting?ids={npc_id}&attackable=true"
339
340                try:
341                    response = requests.get(endpoint, timeout=0.2)
342                    response.raise_for_status()
343                    data = response.json()
344
345                    # If we get a valid NPC response (not error/state), then an NPC is interacting
346                    if data and not data.get("error") and data.get("state") != "Null":
347                        return True
348
349                except requests.RequestException:
350                    continue
351        else:
352            # Prefer the generalized endpoint for any interacting NPC when no specific IDs are provided
353            try:
354                endpoint = f"{self.base_url}npcs/interacting_with_me?attackable=true"
355                response = requests.get(endpoint, timeout=0.3)
356                response.raise_for_status()
357                data = response.json()
358                if data and not data.get("error") and data.get("state") != "Null":
359                    return True
360            except requests.RequestException:
361                pass
362
363        return False
