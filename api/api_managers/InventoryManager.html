186    def interact_with_item(self, item_id: Union[int, List[int]], action: str) -> dict:
187        """
188        Interacts with an inventory item using the specified action.
189        If a list of item IDs is provided, tries each one until one succeeds.
190
191        Args:
192            item_id (Union[int, List[int]]): ID(s) of the item(s) to interact with
193            action (str): Action to perform (e.g., "eat", "drink", "use", etc.)
194
195        Returns:
196            dict: Response from the API indicating success or failure.
197                  For lists, returns success result of first successful interaction,
198                  or error if all attempts failed.
199        """
200        # Ensure action is a string and not empty
201        if not action or not isinstance(action, str):
202            return {"error": f"Invalid action: {action}. Must be a non-empty string."}
203
204        # Handle lists by trying each item until one succeeds
205        if isinstance(item_id, list):
206            if not item_id:
207                return {"error": "Empty item list provided"}
208
209            last_error = None
210            for item in item_id:
211                # First check if we have this item
212                if self.get_item_count(item) > 0:
213                    result = self._interact_with_single_item(item, action)
214                    # If successful (no error), return the success result
215                    if not result.get("error"):
216                        return result
217                    last_error = result
218
219            # If we get here, all attempts failed
220            return last_error or {"error": "No matching items found in inventory"}
221
222        # Handle single item
223        return self._interact_with_single_item(item_id, action)

225    def _interact_with_single_item(self, item_id: int, action: str) -> dict:
226        """
227        Helper method to interact with a single item. This is used so we can pass a list of items to interact with.
228
229        Args:
230            item_id (int): ID of the item to interact with
231            action (str): Action to perform
232
233        Returns:
234            dict: Response from the API
235        """
236        # Ensure item_id is an integer
237        if not isinstance(item_id, int):
238            try:
239                item_id = int(item_id)
240            except (ValueError, TypeError):
241                return {"error": f"Invalid item_id: {item_id}. Must be an integer."}
242
243        endpoint = f"{self.base_url}invokes/inventory/interact_item"
244
245        # Send id as string since Java uses Integer.parseInt()
246        # Always consume clicks by default
247        data = {"id": str(item_id), "action": action.strip(), "consume": "true"}
248
249        try:
250            print(f"Sending inventory interaction request - ID: {item_id}, Action: '{action}', Data: {data}")
251            response = requests.post(endpoint, json=data, timeout=self.timeout)
252
253            # Print response details for debugging
254            print(f"Response status: {response.status_code}")
255            if response.status_code != 200:
256                print(f"Response content: {response.text}")
257
258            response.raise_for_status()
259            result = response.json()
260
261            return result
262        except requests.RequestException as e:
263            print(f"Error in inventory item interaction request: {e}")
264            print(f"Request details - URL: {endpoint}, Data: {data}")
265            return {"error": str(e)}

432    def get_item_count(self, item_id: Union[int, str, List[Union[int, str]]] = None) -> int:
433        """
434        Convenience method to get item count in inventory.
435        If item_id is provided, returns count of that specific item or total count of items in list.
436        If no item_id is provided, returns total count of all items.
437
438        Args:
439            item_id (Union[int, str, List[Union[int, str]]], optional): ID(s) or name(s) of the item(s) to count
440
441        Returns:
442            int: Count of the specified item(s), or total items if no item_id provided
443        """
444        if item_id is None:
445            # Return total item count
446            result = self.is_inventory_full()
447            return int(result.get("totalCount", 0))
448
449        # Handle lists - return total count of all items in the list
450        if isinstance(item_id, list):
451            if not item_id:  # Empty list
452                return 0
453
454            # Separate strings and integers
455            item_names = [item for item in item_id if isinstance(item, str) and not str(item).isdigit()]
456            item_ids = [item for item in item_id if isinstance(item, int)]
457            # Convert string numbers to integers
458            item_ids.extend([int(item) for item in item_id if isinstance(item, str) and str(item).isdigit()])
459
460            # Get items using inventory_contains (which properly formats the request)
461            result = self.inventory_contains(item_names=item_names if item_names else None, item_ids=item_ids if item_ids else None)
462
463            # Sum up all the item counts
464            total_count = 0
465            for item_data in result.values():
466                if isinstance(item_data, dict):
467                    count = item_data.get("itemCount", 0)
468                    if count:
469                        with contextlib.suppress(ValueError, TypeError):
470                            total_count += int(count)
471            return total_count
472
473        # Handle single items
474        if isinstance(item_id, str) and not str(item_id).isdigit():
475            result = self.inventory_contains(item_names=[item_id])
476            return int(result.get(item_id, {}).get("itemCount", 0))
477        else:
478            # Convert to int if it's a string number
479            if isinstance(item_id, str):
480                item_id = int(item_id)
481            result = self.inventory_contains(item_ids=[item_id])
482            return int(result.get(str(item_id), {}).get("itemCount", 0))

484    def get_inventory_item_ids(self) -> List[int]:
485        """
486        Gets all item IDs currently in the inventory, excluding empty slots (-1) and special items (6512).
487
488        Returns:
489            List[int]: List of item IDs in inventory
490        """
491        items = self.get_all_inventory_items()
492        item_ids = []
493
494        for item_data in items.values():
495            if isinstance(item_data, dict):
496                item_id = item_data.get("itemId")
497                if item_id and item_id != -1 and item_id != 6512:
498                    try:
499                        item_ids.append(int(item_id))
500                    except (ValueError, TypeError):
501                        continue
502
503        return item_ids

433    def get_item_count(self, item_id: Union[int, str, List[Union[int, str]]] = None) -> int:
434        """
435        Convenience method to get item count in inventory.
436        If item_id is provided, returns count of that specific item or total count of items in list.
437        If no item_id is provided, returns total count of all items.
438
439        Args:
440            item_id (Union[int, str, List[Union[int, str]]], optional): ID(s) or name(s) of the item(s) to count
441
442        Returns:
443            int: Count of the specified item(s), or total items if no item_id provided
444        """
445        if item_id is None:
446            # Return total item count
447            result = self.is_inventory_full()
448            return int(result.get("totalCount", 0))
449
450        # Handle lists - return total count of all items in the list
451        if isinstance(item_id, list):
452            if not item_id:  # Empty list
453                return 0
454
455            # Separate strings and integers
456            item_names = [item for item in item_id if isinstance(item, str) and not str(item).isdigit()]
457            item_ids = [item for item in item_id if isinstance(item, int)]
458            # Convert string numbers to integers
459            item_ids.extend([int(item) for item in item_id if isinstance(item, str) and str(item).isdigit()])
460
461            # Get items using inventory_contains (which properly formats the request)
462            result = self.inventory_contains(item_names=item_names if item_names else None, item_ids=item_ids if item_ids else None)
463
464            # Sum up all the item counts
465            total_count = 0
466            for item_data in result.values():
467                if isinstance(item_data, dict):
468                    count = item_data.get("itemCount", 0)
469                    if count:
470                        with contextlib.suppress(ValueError, TypeError):
471                            total_count += int(count)
472            return total_count
473
474        # Handle single items
475        if isinstance(item_id, str) and not str(item_id).isdigit():
476            result = self.inventory_contains(item_names=[item_id])
477            return int(result.get(item_id, {}).get("itemCount", 0))
478        else:
479            # Convert to int if it's a string number
480            if isinstance(item_id, str):
481                item_id = int(item_id)
482            result = self.inventory_contains(item_ids=[item_id])
483            return int(result.get(str(item_id), {}).get("itemCount", 0))
