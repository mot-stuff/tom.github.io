171    def handle_dialog(self, choices: list[int]):
172        """
173        Handle dialog interactions by either continuing with space or selecting choices.
174        Uses get_player_dialog_options to verify when we have actual choice options or need to continue.
175
176        Args:
177            choices: List of dialogue choice indices to select
178        """
179        # Keep handling dialog until all choices are made and no more dialog exists
180        remaining_choices = choices.copy() if choices else []
181        last_space_press = 0
182        last_choice_made = 0
183        choice_cooldown = 1.5  # Increased cooldown between choices
184        print(f"Starting dialog handler with choices: {remaining_choices}")
185
186        consecutive_same_state = 0
187        last_dialog_state = None
188        max_consecutive = 5  # Prevent infinite loops
189
190        while True:
191            # Check dialog state using interface API
192            dialog_data = self.api_m.interfaces.get_player_dialog_options()
193            has_dialog = dialog_data.get("hasDialog") in [True, "True"]  # Handle both boolean and string
194            dialog_options = dialog_data.get("options", [])
195            current_state = (has_dialog, len(dialog_options), tuple(dialog_options) if dialog_options else ())
196
197            # Track consecutive same states to prevent infinite loops
198            if current_state == last_dialog_state:
199                consecutive_same_state += 1
200                if consecutive_same_state >= max_consecutive:
201                    print(f"Breaking due to consecutive same state ({consecutive_same_state}x): {current_state}")
202                    break
203            else:
204                consecutive_same_state = 0
205                last_dialog_state = current_state
206
207            print(f"Dialog state - has_dialog: {has_dialog}, options: {dialog_options}")
208
209            if not has_dialog:
210                print("No dialog detected, checking again...")
211                # No dialog at all, wait briefly then check again
212                time.sleep(0.5)  # Increased wait time
213                # Check one more time before giving up
214                dialog_data = self.api_m.interfaces.get_player_dialog_options()
215                has_dialog = dialog_data.get("hasDialog") in [True, "True"]  # Handle both boolean and string
216                if not has_dialog:
217                    print("Still no dialog, breaking out")
218                    break
219                print("Dialog found on second check")
220                continue
221
222            # Check if we have "Click here to continue" or options
223            if len(dialog_options) == 1 and "continue" in dialog_options[0].lower():
224                print("Found 'Click here to continue' dialog")
225                # Need to continue through dialog
226                current_time = time.time()
227                if (current_time - last_space_press) > 0.8:  # Increased cooldown
228                    print("Sending Space key")
229                    self.api_m.controller.send_key("Space")
230                    last_space_press = current_time
231                    time.sleep(1.0)  # Longer wait for dialog to update
232            elif len(dialog_options) > 1 and remaining_choices:
233                print(f"Found choice dialog with options: {dialog_options}")
234                current_time = time.time()
235                # Ensure we don't make choices too rapidly
236                if (current_time - last_choice_made) > choice_cooldown:
237                    # We have options to choose from
238                    choice = remaining_choices.pop(0)
239                    print(f"Selecting choice {choice} from remaining: {remaining_choices}")
240                    self.api_m.controller.send_key(str(choice))
241                    last_choice_made = current_time
242                    time.sleep(1.2)  # Longer wait for choice to register and dialog to change
243                else:
244                    print("Choice cooldown active, waiting...")
245                    time.sleep(0.3)
246            elif len(dialog_options) > 1 and not remaining_choices:
247                print("Found choice dialog but no more choices to make - breaking out")
248                break
249            else:
250                print(f"Unexpected dialog state - options: {dialog_options}, remaining choices: {remaining_choices}")
251                # Try space as fallback
252                current_time = time.time()
253                if (current_time - last_space_press) > 0.8:
254                    print("Fallback: sending Space key")
255                    self.api_m.controller.send_key("Space")
256                    last_space_press = current_time
257                    time.sleep(1.0)
258
259            # Longer delay between checks to allow dialog transitions
260            time.sleep(0.5)

309                time.sleep(0.6)  # Wait for choice to register
310            else:
311                print(f"Unexpected dialog state - options: {dialog_options}")
312
313            # Small delay between checks
314            time.sleep(0.3)

329    def handle_dialog_with_text_parsing(self, text_patterns: dict) -> int:
330        """
331        Handle dialog interactions while monitoring for specific text patterns.
332        Used for special quest dialog where we need to extract information from the NPC's speech.
333
334        Args:
335            text_patterns (dict): Dictionary mapping text patterns to return values
336                                Example: {"Less than my hands": 1, "More than my head less than my fingers": 2}
337
338        Returns:
339            int: The value associated with the matched pattern, or -1 if no pattern matched
340        """
341        print(f"Starting special dialog handler looking for patterns: {list(text_patterns.keys())}")
342        last_space_press = 0
343        found_value = -1
344
345        def safe_encode_text(text: str) -> str:
346            """Safely encode text to handle unicode characters"""
347            try:
348                # Remove or replace problematic unicode characters
349                return text.encode("ascii", errors="ignore").decode("ascii")
350            except (UnicodeEncodeError, UnicodeDecodeError, AttributeError):
351                return str(text)
352
353        while True:
354            # Check dialog state
355            dialog_data = self.api_m.interfaces.get_player_dialog_options()
356            has_dialog = dialog_data.get("hasDialog") in [True, "True"]
357            dialog_options = dialog_data.get("options", [])
358
359            if not has_dialog:
360                print("No dialog detected, checking for chat messages...")
361                # Check if we have found our pattern value
362                if found_value != -1:
363                    print(f"Pattern matched, returning value: {found_value}")
364                    return found_value
365
366                # No dialog, wait briefly then check again
367                time.sleep(0.3)
368                dialog_data = self.api_m.interfaces.get_player_dialog_options()
369                has_dialog = dialog_data.get("hasDialog") in [True, "True"]
370                if not has_dialog:
371                    print("Still no dialog, breaking out")
372                    break
373                continue
374
375            # Check latest chat message for our patterns
376            try:
377                chat_data = self.api_m.get_latest_chat_message()
378                if chat_data:
379                    message = chat_data.get("lastMessage", "")
380                    message_type = chat_data.get("type", "")
381
382                    # Safely encode the message to handle unicode
383                    safe_message = safe_encode_text(message)
384                    safe_message_type = safe_encode_text(message_type)
385
386                    print(f"Checking message: '{safe_message}' (type: {safe_message_type})")
387
388                    # Look for our patterns in the message
389                    for pattern, value in text_patterns.items():
390                        # Convert pattern to key words for loose matching
391                        pattern_words = pattern.lower().split()
392                        safe_message_lower = safe_message.lower()
393
394                        # Check if all key words from the pattern are present
395                        if all(word in safe_message_lower for word in pattern_words):
396                            print(f"Found pattern '{pattern}' (loose match) in message, value: {value}")
397                            found_value = value
398                            # Continue dialog to finish the conversation
399                            break
400            except (ValueError, KeyError, TypeError) as e:
401                print(f"Error processing chat message: {e}")
402                # Continue with dialog handling even if we can't read the message
403
404            # Handle dialog continuation
405            if len(dialog_options) == 1 and "continue" in dialog_options[0].lower():
406                print("Found 'Click here to continue' dialog")
407                current_time = time.time()
408                if (current_time - last_space_press) > 0.6:
409                    print("Sending Space key")
410                    self.api_m.controller.send_key("Space")
411                    last_space_press = current_time
412                    time.sleep(0.6)
413            elif len(dialog_options) > 1:
414                print("Found choice dialog - selecting first option to continue")
415                self.api_m.controller.send_key("1")
416                time.sleep(0.6)
417            else:
418                print("Unexpected dialog state, continuing...")
419                current_time = time.time()
420                if (current_time - last_space_press) > 0.6:
421                    self.api_m.controller.send_key("Space")
422                    last_space_press = current_time
423                    time.sleep(0.6)
424
425            time.sleep(0.3)
426
427        return found_value

330    def handle_dialog_with_text_parsing(self, text_patterns: dict) -> int:
331        """
332        Handle dialog interactions while monitoring for specific text patterns.
333        Used for special quest dialog where we need to extract information from the NPC's speech.
334
335        Args:
336            text_patterns (dict): Dictionary mapping text patterns to return values
337                                Example: {"Less than my hands": 1, "More than my head less than my fingers": 2}
338
339        Returns:
340            int: The value associated with the matched pattern, or -1 if no pattern matched
341        """
342        print(f"Starting special dialog handler looking for patterns: {list(text_patterns.keys())}")
343        last_space_press = 0
344        found_value = -1
345
346        def safe_encode_text(text: str) -> str:
347            """Safely encode text to handle unicode characters"""
348            try:
349                # Remove or replace problematic unicode characters
350                return text.encode("ascii", errors="ignore").decode("ascii")
351            except (UnicodeEncodeError, UnicodeDecodeError, AttributeError):
352                return str(text)
353
354        while True:
355            # Check dialog state
356            dialog_data = self.api_m.interfaces.get_player_dialog_options()
357            has_dialog = dialog_data.get("hasDialog") in [True, "True"]
358            dialog_options = dialog_data.get("options", [])
359
360            if not has_dialog:
361                print("No dialog detected, checking for chat messages...")
362                # Check if we have found our pattern value
363                if found_value != -1:
364                    print(f"Pattern matched, returning value: {found_value}")
365                    return found_value
366
367                # No dialog, wait briefly then check again
368                time.sleep(0.3)
369                dialog_data = self.api_m.interfaces.get_player_dialog_options()
370                has_dialog = dialog_data.get("hasDialog") in [True, "True"]
371                if not has_dialog:
372                    print("Still no dialog, breaking out")
373                    break
374                continue
375
376            # Check latest chat message for our patterns
377            try:
378                chat_data = self.api_m.get_latest_chat_message()
379                if chat_data:
380                    message = chat_data.get("lastMessage", "")
381                    message_type = chat_data.get("type", "")
382
383                    # Safely encode the message to handle unicode
384                    safe_message = safe_encode_text(message)
385                    safe_message_type = safe_encode_text(message_type)
386
387                    print(f"Checking message: '{safe_message}' (type: {safe_message_type})")
388
389                    # Look for our patterns in the message
390                    for pattern, value in text_patterns.items():
391                        # Convert pattern to key words for loose matching
392                        pattern_words = pattern.lower().split()
393                        safe_message_lower = safe_message.lower()
394
395                        # Check if all key words from the pattern are present
396                        if all(word in safe_message_lower for word in pattern_words):
397                            print(f"Found pattern '{pattern}' (loose match) in message, value: {value}")
398                            found_value = value
399                            # Continue dialog to finish the conversation
400                            break
401            except (ValueError, KeyError, TypeError) as e:
402                print(f"Error processing chat message: {e}")
403                # Continue with dialog handling even if we can't read the message
404
405            # Handle dialog continuation
406            if len(dialog_options) == 1 and "continue" in dialog_options[0].lower():
407                print("Found 'Click here to continue' dialog")
408                current_time = time.time()
409                if (current_time - last_space_press) > 0.6:
410                    print("Sending Space key")
411                    self.api_m.controller.send_key("Space")
412                    last_space_press = current_time
413                    time.sleep(0.6)
414            elif len(dialog_options) > 1:
415                print("Found choice dialog - selecting first option to continue")
416                self.api_m.controller.send_key("1")
417                time.sleep(0.6)
418            else:
419                print("Unexpected dialog state, continuing...")
420                current_time = time.time()
421                if (current_time - last_space_press) > 0.6:
422                    self.api_m.controller.send_key("Space")
423                    last_space_press = current_time
424                    time.sleep(0.6)
425
426            time.sleep(0.3)
427
428        return found_value
