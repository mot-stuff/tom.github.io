 11    def get_tracked_player_position(self) -> Optional[Tuple[int, int]]:
 12        """
 13        Gets the screen position of the currently tracked player.
 14        Since the overlay handles tracking automatically, we'll try to get a nearby player's position.
 15
 16        Returns:
 17            Tuple[int, int]: Screen coordinates (x, y) of a nearby player, or None if no players found
 18        """
 19        try:
 20            # Get nearby players within reasonable distance (15 tiles as per overlay logic)
 21            nearby_players = self.api_m.players_api.get_players_within_distance(15)
 22
 23            if nearby_players and len(nearby_players) > 0:
 24                # Get the first player's bounds for position
 25                for _player_name, player_data in nearby_players.items():
 26                    player_bounds = player_data.get("playerBounds", {})
 27                    centerx = player_bounds.get("centerX")
 28                    centery = player_bounds.get("centerY")
 29                    if centerx is not None and centery is not None:
 30                        return (int(centerx), int(centery))
 31
 32            return None
 33        except Exception:
 34            return None

171    def get_target_combat_level_range(self, my_combat_level: int, wilderness_level: int) -> Tuple[int, int]:
172        """
173        Calculates the valid combat level range for targeting in the wilderness.
174
175        Args:

177            wilderness_level (int): Current wilderness level

179        Returns:
180            Tuple[int, int]: (min_level, max_level) that you can attack

186    def find_attackable_players(self, my_combat_level: int, wilderness_level: int, max_distance: int = 15) -> Optional[Dict]:
187        """
188        Finds all players within attack range based on combat level and distance.
189
190        Args:
191            my_combat_level (int): Your combat level
192            wilderness_level (int): Current wilderness level
193            max_distance (int): Maximum distance to search for players
194
195        Returns:
196            Optional[Dict]: Dictionary of attackable players
197        """
198        min_level, max_level = self.get_target_combat_level_range(my_combat_level, wilderness_level)
199
200        # Get players within combat level range
201        level_filtered = self.get_nearby_players_by_combat_level(min_level, max_level)
202        if not level_filtered:
203            return None
204
205        # Get players within distance
206        distance_filtered = self.get_nearby_players_by_distance(max_distance)
207        if not distance_filtered:
208            return None
209
210        # Return intersection of both filters
211        attackable = {}
212        for player_name in level_filtered:
213            if player_name in distance_filtered:
214                attackable[player_name] = level_filtered[player_name]
215
216        return attackable if attackable else None

249    def check_players_at_spot_for_duration(self, target_x: int, target_y: int, target_z: int, distance: int = 5, duration_threshold: float = 4.0) -> bool:
250        """
251        Checks if players have been at a specific spot for more than the duration threshold.
252        This is useful for detecting if a training spot is occupied.
253
254        Args:
255            target_x (int): Target X coordinate
256            target_y (int): Target Y coordinate
257            target_z (int): Target Z coordinate (plane)
258            distance (int): Maximum distance in tiles from target location
259            duration_threshold (float): How long players need to be there before returning True (seconds)
260
261        Returns:
262            bool: True if players have been at the spot for longer than duration_threshold, False otherwise
263        """
264        import time
265
266        # Initialize tracking if not already done
267        if not hasattr(self, "_spot_tracking"):
268            self._spot_tracking = {}
269
270        spot_key = f"{target_x},{target_y},{target_z}"
271        current_time = time.time()
272
273        # Get current players at the spot
274        players_at_spot = self.get_players_at_location(target_x, target_y, target_z, distance)
275
276        if players_at_spot and len(players_at_spot) > 0:
277            # Players are at the spot
278            if spot_key not in self._spot_tracking:
279                # First time seeing players here
280                self._spot_tracking[spot_key] = current_time
281                return False
282            else:
283                # Check how long they've been here
284                time_at_spot = current_time - self._spot_tracking[spot_key]
285                return time_at_spot >= duration_threshold
286        else:
287            # No players at the spot, reset tracking
288            if spot_key in self._spot_tracking:
289                del self._spot_tracking[spot_key]
290            return False

187    def find_attackable_players(self, my_combat_level: int, wilderness_level: int, max_distance: int = 15) -> Optional[Dict]:
188        """

190
191        Args:
192            my_combat_level (int): Your combat level
193            wilderness_level (int): Current wilderness level
194            max_distance (int): Maximum distance to search for players

197            Optional[Dict]: Dictionary of attackable players

def check_players_at_spot_for_duration( self, target_x: int, target_y: int, target_z: int, distance: int = 5, duration_threshold: float = 4.0) -> bool: 
250    def check_players_at_spot_for_duration(self, target_x: int, target_y: int, target_z: int, distance: int = 5, duration_threshold: float = 4.0) -> bool:
251        """
252        Checks if players have been at a specific spot for more than the duration threshold.
253        This is useful for detecting if a training spot is occupied.
254
255        Args:
256            target_x (int): Target X coordinate
257            target_y (int): Target Y coordinate
258            target_z (int): Target Z coordinate (plane)
259            distance (int): Maximum distance in tiles from target location
260            duration_threshold (float): How long players need to be there before returning True (seconds)
261
262        Returns:
263            bool: True if players have been at the spot for longer than duration_threshold, False otherwise
264        """
265        import time
266
267        # Initialize tracking if not already done
268        if not hasattr(self, "_spot_tracking"):
269            self._spot_tracking = {}
270
271        spot_key = f"{target_x},{target_y},{target_z}"
272        current_time = time.time()
273
274        # Get current players at the spot
275        players_at_spot = self.get_players_at_location(target_x, target_y, target_z, distance)
276
277        if players_at_spot and len(players_at_spot) > 0:
278            # Players are at the spot
279            if spot_key not in self._spot_tracking:
280                # First time seeing players here
281                self._spot_tracking[spot_key] = current_time
282                return False
283            else:
284                # Check how long they've been here
285                time_at_spot = current_time - self._spot_tracking[spot_key]
286                return time_at_spot >= duration_threshold
287        else:
288            # No players at the spot, reset tracking
289            if spot_key in self._spot_tracking:
290                del self._spot_tracking[spot_key]
291            return False
