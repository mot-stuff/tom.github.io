354        """
355        Performs a "misclick" by walking to an adjacent tile instead of clicking the NPC.
356
357        Args:
358            npc_location: Tuple of (x, y, z) coordinates of the target NPC
359
360        Returns:
361            bool: True if misclick was performed, False if it failed
362        """
363        try:
364            if self.bot_instance:
365                self.bot_instance.log_msg(f"NPC misclick attempt: location={npc_location}")
366
367            if not npc_location or not all(isinstance(coord, (int, float)) for coord in npc_location):
368                if self.bot_instance:
369                    self.bot_instance.log_msg("NPC misclick failed: Invalid NPC location")
370                return False
371
372            # Get adjacent navigable tiles
373            adjacent_tiles = self.get_adjacent_navigable_tiles(npc_location)
374
375            if not adjacent_tiles:
376                if self.bot_instance:
377                    self.bot_instance.log_msg("NPC misclick failed: No adjacent tiles")
378                return False
379
380            # Select a random adjacent tile
381            misclick_target = random.choice(adjacent_tiles)
382
383            if self.bot_instance:
384                self.bot_instance.log_msg(f"Misclicking NPC to adjacent tile: {misclick_target[:2]}")
385
386            # Walk to the adjacent tile (this simulates the "miss")
387            success = self.api_m.movement.walk_to_world_point(misclick_target)
388
389            if not success:
390                if self.bot_instance:

392                return False
393
394            if self.bot_instance:
395                self.bot_instance.log_msg("NPC misclick successful!")
396
397            # Add a brief pause to simulate realizing the mistake
398            mistake_realization_delay = rd.truncated_normal_sample(0.3, 1.2)

402
403        except Exception as e:

539
540        # Remove snapshots older than 6 ticks
541        self.xp_snapshots = [(timestamp, xp_dict) for timestamp, xp_dict in self.xp_snapshots if current_time - timestamp <= tick_window_seconds]

544        current_xp = {}

672        except Exception as e:
673            if self.bot_instance:
