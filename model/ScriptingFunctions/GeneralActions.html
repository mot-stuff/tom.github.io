 17    def take_break(self, min_seconds: int = 1, max_seconds: int = 30, fancy: bool = False):
 18        """
 19        Takes a break for a random amount of time.
 20        Args:
 21            min_seconds: minimum amount of time the bot could rest
 22            max_seconds: maximum amount of time the bot could rest
 23            fancy: if True, the randomly generated value will be from a truncated normal distribution
 24                   with randomly selected means. This may produce more human results.
 25        """
 26        if fancy:
 27            length = rd.fancy_normal_sample(min_seconds, max_seconds)
 28        else:
 29            length = rd.truncated_normal_sample(min_seconds, max_seconds)
 30        length = round(length)
 31        for _ in range(length):
 32            time.sleep(1)

172            stat_names.CONSTRUCTION,

246        Args:

251        """

253            stat_names.ATTACK,
254            stat_names.DEFENCE,

256            stat_names.HITPOINTS,

258            stat_names.PRAYER,
259            stat_names.MAGIC,

268            stat_names.HERBLORE,
269            stat_names.AGILITY,

271            stat_names.SLAYER,
272            stat_names.FARMING,
273            stat_names.RUNECRAFTING,
274            stat_names.HUNTER,
275            stat_names.CONSTRUCTION,
276        ]
277
278        # If no snapshot provided, take current and return empty (nothing to compare)

316                self.log_msg(f"Attack successful - gained {gained_skill} XP")
317        """
318        # If no snapshot provided, take current and return empty (nothing to compare)
