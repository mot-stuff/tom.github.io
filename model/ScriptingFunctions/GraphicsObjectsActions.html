 15    def get_dangerous_tiles(self, dangerous_object_ids: List[int] = None) -> List[Tuple[int, int, int]]:
 16        """
 17        Get a list of dangerous tile coordinates from graphics objects.
 18
 19        Args:
 20            dangerous_object_ids: List of graphics object IDs that represent dangerous areas.
 21                                If None, uses a default set for common dangerous graphics.
 22
 23        Returns:
 24            List[Tuple[int, int, int]]: List of (x, y, z) coordinates of dangerous tiles
 25        """
 26        if dangerous_object_ids is None:
 27            # Default dangerous graphics object IDs - should be updated based on specific bosses
 28            dangerous_object_ids = []
 29
 30        dangerous_tiles = []
 31        print(f"[DEBUG] Looking for dangerous graphics objects with IDs: {dangerous_object_ids}")
 32
 33        try:
 34            # Get all graphics objects
 35            all_objects = self.api_m.graphics_objects.get_all_graphics_objects()
 36            print(f"[DEBUG] Graphics API returned {len(all_objects) if all_objects else 0} object types")
 37
 38            if not all_objects:
 39                print("[DEBUG] No graphics objects found - API returned empty/None")
 40                return dangerous_tiles
 41
 42            # Process each graphics object
 43            for object_id_str, objects_list in all_objects.items():
 44                try:
 45                    object_id = int(object_id_str)
 46
 47                    # Check if this is a dangerous object type
 48                    if object_id in dangerous_object_ids:
 49                        print(f"[DEBUG] Found dangerous graphics object ID {object_id}!")
 50                        if isinstance(objects_list, list):
 51                            print(f"[DEBUG] Processing {len(objects_list)} instances of object {object_id}")
 52                            for i, obj in enumerate(objects_list):
 53                                if "graphicsObjectWorldLocation" in obj:
 54                                    location = obj["graphicsObjectWorldLocation"]
 55                                    if all(coord in location for coord in ["locationX", "locationY", "locationZ"]):
 56                                        tile = (location["locationX"], location["locationY"], location["locationZ"])
 57                                        dangerous_tiles.append(tile)
 58                                        print(f"[DEBUG] Added dangerous tile [{i+1}]: {tile}")
 59                                    else:
 60                                        print(f"[DEBUG] Object {i+1} missing locationX/locationY/locationZ coordinates: {location}")
 61                                else:
 62                                    print(f"[DEBUG] Object {i+1} missing world location data")
 63                        else:
 64                            print(f"[DEBUG] Object {object_id} data is not a list: {type(objects_list)}")
 65                except (ValueError, TypeError) as e:
 66                    if self.bot_instance:
 67                        self.bot_instance.log_msg(f"Error processing graphics object {object_id_str}: {e}")
 68                    print(f"[DEBUG] Error processing object {object_id_str}: {e}")
 69                    continue
 70
 71        except Exception as e:
 72            if self.bot_instance:
 73                self.bot_instance.log_msg(f"Error getting dangerous tiles: {e}")
 74            print(f"[DEBUG] Error in get_dangerous_tiles: {e}")
 75
 76        print(f"[DEBUG] Final result: {len(dangerous_tiles)} dangerous tiles found")
 77        return dangerous_tiles

 79    def is_player_on_dangerous_tile(self, player_location: Tuple[int, int, int] = None, dangerous_object_ids: List[int] = None) -> bool:
 80        """
 81        Check if the player is currently standing on a dangerous tile.
 82
 83        Args:
 84            player_location: Player's current location. If None, gets current position.
 85            dangerous_object_ids: List of dangerous graphics object IDs.
 86
 87        Returns:
 88            bool: True if player is on a dangerous tile, False otherwise
 89        """
 90        try:
 91            if player_location is None:
 92                player_location = self.api_m.get_player_position()
 93
 94            print(f"[DEBUG] Checking if player at {player_location} is on dangerous tile...")
 95
 96            dangerous_tiles = self.get_dangerous_tiles(dangerous_object_ids)
 97            print(f"[DEBUG] Found {len(dangerous_tiles)} dangerous tiles: {dangerous_tiles}")
 98
 99            # Check if player location matches any dangerous tile
100            for i, dangerous_tile in enumerate(dangerous_tiles):
101                print(f"[DEBUG] Comparing player {player_location} with dangerous tile [{i+1}] {dangerous_tile}")
102                if player_location[0] == dangerous_tile[0] and player_location[1] == dangerous_tile[1] and player_location[2] == dangerous_tile[2]:
103                    print(f"[DEBUG] MATCH FOUND! Player is on dangerous tile {dangerous_tile}")
104                    return True
105
106            print("[DEBUG] No exact matches found. Player is safe.")
107
108        except Exception as e:
109            if self.bot_instance:
110                self.bot_instance.log_msg(f"Error checking if player on dangerous tile: {e}")
111            print(f"[DEBUG] Error in is_player_on_dangerous_tile: {e}")
112
113        return False

115    def get_safe_adjacent_tile(
116        self, player_location: Tuple[int, int, int], dangerous_tiles: List[Tuple[int, int, int]] = None, max_distance: int = 4
117    ) -> Optional[Tuple[int, int, int]]:
118        """
119        Find a safe tile adjacent to the player's current position.
120
121        Args:
122            player_location: Player's current location (x, y, z)
123            dangerous_tiles: List of dangerous tile coordinates
124            max_distance: Maximum distance to search for safe tiles
125
126        Returns:
127            Optional[Tuple[int, int, int]]: Safe tile coordinates or None if no safe tile found
128        """
129        if dangerous_tiles is None:
130            dangerous_tiles = []
131
132        try:
133            x, y, z = player_location
134            print(f"[DEBUG] Looking for safe tile around {player_location}")
135            print(f"[DEBUG] Dangerous tiles to avoid: {dangerous_tiles}")
136
137            # Check tiles in expanding radius
138            for distance in range(1, max_distance + 1):
139                print(f"[DEBUG] Checking distance {distance} from player")
140                # Generate potential safe tiles around the player
141                potential_tiles = []
142
143                for dx in range(-distance, distance + 1):
144                    for dy in range(-distance, distance + 1):
145                        if abs(dx) == distance or abs(dy) == distance:  # Only check perimeter
146                            potential_tiles.append((x + dx, y + dy, z))
147
148                print(f"[DEBUG] Generated {len(potential_tiles)} potential tiles at distance {distance}: {potential_tiles}")
149
150                # Shuffle to randomize selection
151                random.shuffle(potential_tiles)
152
153                # Check each potential tile
154                for tile in potential_tiles:
155                    print(f"[DEBUG] Checking if tile {tile} is safe...")
156                    if tile not in dangerous_tiles:
157                        print(f"[DEBUG] Found safe tile: {tile}")
158                        return tile
159                    else:
160                        print(f"[DEBUG] Tile {tile} is dangerous, skipping")
161
162                print(f"[DEBUG] No safe tiles found at distance {distance}")
163
164        except Exception as e:
165            if self.bot_instance:
166                self.bot_instance.log_msg(f"Error finding safe tile: {e}")
167            print(f"[DEBUG] Exception in get_safe_adjacent_tile: {e}")
168
169        print(f"[DEBUG] No safe tile found within {max_distance} tiles of {player_location}")
170        return None
