 96    def pathfind_to(self, coords, z=None, ignore_minimum=False):
 97        """
 98        Main pathfinding method using walker API with comprehensive features:
 99        - Automatic door/obstacle handling
100        - Transport detection and usage
101        - Teleport item integration
102        - Human-like movement patterns
103
104        Pass the World X and World Y or a tuple of the destination to pathfind to it.
105        Uses the most advanced pathfinding system available.
106
107        This method will wait until the player reaches the destination or stops moving.
108        """
109        while not self.api_m.general_api.get_is_player_idle():
110            time.sleep(0.2)
111        self.check_and_use_stamina()
112
113        # Handle legacy z parameter
114        if z is not None:
115            if isinstance(coords, tuple):
116                # If coords is already a tuple, just add z to it
117                coords = (coords[0], coords[1], z)
118            else:
119                # If coords is a single value, create tuple with z
120                coords = (coords, z)
121
122        if isinstance(coords, tuple):
123            if len(coords) == 2:
124                x, y = coords
125                current_x, current_y, current_z = self.api_m.get_player_position()
126                z = current_z
127            else:
128                x, y, z = coords
129        else:
130            raise ValueError("coords must be a tuple")
131
132        target_world_point = (x, y, z)
133
134        # Check if already at target
135        current_pos = self.api_m.get_player_position()
136        current_distance = abs(current_pos[0] - x) + abs(current_pos[1] - y)
137        if current_distance <= 2:
138            if self.bot_instance:
139                self.bot_instance.log_msg(f"Already at target {target_world_point}")
140            return {"result": "Already at target"}
141
142        # Set pathfinding action state
143        if self.bot_instance:
144            self.bot_instance.set_camera_action_state("pathfinding")
145
146        try:
147            # Start pathfinding (don't wait for completion in the API call)
148            result = self.api_m.walker_walk_to_world_point(target_world_point, max_distance=2, wait_for_completion=False)  # We'll handle waiting ourselves
149            if result.get("error"):
150                if self.bot_instance:
151                    self.bot_instance.log_msg(f"Pathfinding failed: {result.get('error')}")
152                return result
153
154            if self.bot_instance:
155                self.bot_instance.log_msg(f"Pathfinding started to {target_world_point}")
156
157            # Wait for pathfinding to actually start (API takes a few seconds to begin)
158            pathfinding_start_timeout = 5  # Wait up to 10 seconds for pathfinding to start
159            pathfinding_start_time = time.time()
160            initial_position = self.api_m.get_player_position()
161
162            while time.time() - pathfinding_start_time < pathfinding_start_timeout:
163                current_pos = self.api_m.get_player_position()
164                # Check if player has started moving (indicating pathfinding has begun)
165                if current_pos != initial_position:
166                    if self.bot_instance:
167                        self.bot_instance.log_msg("Pathfinding has started - player is moving")
168                    break
169
170                # Also check if player is no longer idle (another indicator that pathfinding started)
171                if not self.api_m.get_is_player_idle():
172                    if self.bot_instance:
173                        self.bot_instance.log_msg("Pathfinding has started - player is no longer idle")
174                    break
175
176                time.sleep(0.5)
177            else:
178                # Timeout waiting for pathfinding to start
179                if self.bot_instance:
180                    self.bot_instance.log_msg("Pathfinding failed to start within timeout period")
181                return {"result": "Pathfinding failed to start"}
182
183            # Wait for pathfinding to complete
184            max_wait_time = 240  # Maximum 4 minutes to reach target
185            start_time = time.time()
186            last_position = None
187            idle_count = 0
188
189            while time.time() - start_time < max_wait_time:
190                current_pos = self.api_m.get_player_position()
191                current_distance = abs(current_pos[0] - x) + abs(current_pos[1] - y)
192
193                # Check if we've reached the target
194                if current_distance <= 2:
195                    if self.bot_instance:
196                        self.bot_instance.log_msg(f"Reached target {target_world_point}")
197                    return {"result": "Successfully reached target"}
198
199                # Check if player is idle/stuck
200                if last_position == current_pos:
201                    idle_count += 1
202                    if idle_count >= 3:  # 3 seconds of no movement
203                        if self.api_m.get_is_player_idle():
204                            if self.bot_instance:
205                                self.bot_instance.log_msg("Player is idle, pathfinding appears complete but not at target")
206                            return {"result": "Pathfinding stopped - player idle"}
207                else:
208                    idle_count = 0
209                    last_position = current_pos
210
211                # Check stamina during pathfinding
212                self.check_and_use_stamina()
213
214                time.sleep(1.0)  # Check every second
215
216            # Timeout reached
217            if self.bot_instance:
218                self.bot_instance.log_msg(f"Pathfinding timeout after {max_wait_time}s")
219            return {"result": "Pathfinding timeout"}
220
221        finally:
222            # Clear pathfinding action state
223            if self.bot_instance:
224                self.bot_instance.set_camera_action_state("idle")

422    def click_minimap_location(self, world_point: Tuple[int, int, int], button: str = "left") -> bool:
423        """
424        Click on a world location using the minimap.
425
426        Args:
427            world_point (Tuple[int, int, int]): World coordinates to click
428            button (str): Mouse button to use ("left", "right", "middle")
429
430        Returns:
431            bool: True if successful, False otherwise
432        """
433        try:
434            # Convert world coordinates to minimap coordinates
435            minimap_result = self.api_m.walker_world_to_minimap(world_point)
436
437            if minimap_result.get("error"):
438                if self.bot_instance:
439                    self.bot_instance.log_msg(f"Failed to convert world point to minimap: {minimap_result.get('error')}")
440                return False
441
442            if minimap_result.get("isVisible") != "True":
443                if self.bot_instance:
444                    self.bot_instance.log_msg(f"World point {world_point} is not visible on minimap")
445                return False
446
447            # Get canvas coordinates
448            canvas_x = minimap_result.get("minimapRelativeX")
449            canvas_y = minimap_result.get("minimapRelativeY")
450
451            if canvas_x is None or canvas_y is None:
452                if self.bot_instance:
453                    self.bot_instance.log_msg("Failed to get canvas coordinates from minimap")
454                return False
455
456            # Use hardware mouse to click on the minimap
457            success = self.mouse.move_to((canvas_x, canvas_y))
458            time.sleep(0.1)
459            self.mouse.click()
460
461            return success
462
463        except Exception as e:
464            if self.bot_instance:
465                self.bot_instance.log_msg(f"Error clicking minimap location: {e}")
466            return False

424        """
425        Click on a world location using the minimap.
426
427        Args:
428            world_point (Tuple[int, int, int]): World coordinates to click
429            button (str): Mouse button to use ("left", "right", "middle")
430
431        Returns:
432            bool: True if successful, False otherwise
433        """
434        try:
435            # Convert world coordinates to minimap coordinates
436            minimap_result = self.api_m.walker_world_to_minimap(world_point)

438            if minimap_result.get("error"):
439                if self.bot_instance:
440                    self.bot_instance.log_msg(f"Failed to convert world point to minimap: {minimap_result.get('error')}")
441                return False
442
443            if minimap_result.get("isVisible") != "True":
444                if self.bot_instance:
445                    self.bot_instance.log_msg(f"World point {world_point} is not visible on minimap")
446                return False
447
448            # Get canvas coordinates
449            canvas_x = minimap_result.get("minimapRelativeX")
450            canvas_y = minimap_result.get("minimapRelativeY")
451
452            if canvas_x is None or canvas_y is None:
453                if self.bot_instance:
454                    self.bot_instance.log_msg("Failed to get canvas coordinates from minimap")
455                return False
456
457            # Use hardware mouse to click on the minimap
458            success = self.mouse.move_to((canvas_x, canvas_y))
459            time.sleep(0.1)
460            self.mouse.click()
461
462            return success
463
464        except Exception as e:
465            if self.bot_instance:
466                self.bot_instance.log_msg(f"Error clicking minimap location: {e}")
467            return False
